# 数据的存储方式

先看一下一个下面的例子

const ADD = 'ADD';
const storeOptions = {
  modules: {
    a: {
      namespaced: true,
      state: () => ({
        nums: [],
      }),
      mutations: {
        ADD(state, data) {
          state.push(data);
        }
      },
      actions: {
        plus({ commit }, data) {
          commit(ADD, data);
        }
      },
      getters: {
        maxNum(state) { return Math.max(...state.nums); }
      },
      modules: {
        b: {
          namespaced: true,
          state: () => ({
            nums: [],
          }),
          mutations: {
            ADD(state, data) {
              state.push(data);
            }
          },
          actions: {
            plus({ commit }, data) {
              commit(ADD, data);
            }
          },
          getters: {
            maxNum(state) { return Math.max(...state.nums); }
          },
          modules: {
            c: {
              namespaced: true,
              state: () => ({
                nums: [],
              }),
              mutations: {
                ADD(state, data) {
                  state.push(data);
                }
              },
              actions: {
                plus({ commit }, data) {
                  commit(ADD, data);
                }
              },
              getters: {
                maxNum(state) { return Math.max(...state.nums); }
              },
            }

          }
        }
      }
    }
  }
}

const store = new window.Vuex.Store(window.storeOptions)
vm = new Vue({
  el: '#container',
  store,
  data() {
    console.log('store', store);
    return {};
  }
})
